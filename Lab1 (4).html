#RLE(run-length encoding) encoder
def encode(string):
    if string == '':
        return ''
    i = 0
    count = 0
    letter = string[i]
    rle = []
    while i <= len(string) - 1:        
        while string[i] == letter:
            i+= 1
            count +=1
            #catch the loop on last character so it doesn't got to top and access out of bounds
            if i > len(string) - 1:
                break
    
        rle.append('{1}{0}'.format(count, letter))
        if i > len(string) - 1: 
            break
        letter = string[i]
        count = 0
    #join list of strings together to create one string to return
    final = ''.join(rle)
    return final
